// Example React logic for your landing page
import { useEffect } from 'react';

function LandingPage() {
  useEffect(() => {
    const urlParams = new URLSearchParams(window.location.search);
    const ref = urlParams.get('ref');
    if (ref) {
      localStorage.setItem('referrer_id', ref);
    }
  }, []);

  const getPlayStoreLink = () => {
    const refId = localStorage.getItem('referrer_id') || 'organic';
    // Encodes the referrer directly for the Google Play tracking API
    return `https://google.com{refId}`;
  };

  return (
    <a href={getPlayStoreLink()} className="bg-green-600 text-white font-bold py-3 px-6 rounded-lg hover:bg-green-700 transition">
      Download Android App
    </a>
  );
}

2. Android App (Flutter or Native Kotlin)
You need to extract that utm_campaign variable when the user launches the app for the very first time.
If using Flutter: Use the pub.dev android_play_install_referrer package to read the Google Play Store tracking token securely without setting up deep links.

// Native Kotlin implementation inside your MainActivity or LaunchActivity
import com.android.installreferrer.api.InstallReferrerClient
import com.android.installreferrer.api.InstallReferrerStateListener
import com.android.installreferrer.api.ReferrerDetails

val referrerClient = InstallReferrerClient.newBuilder(this).build()
referrerClient.startConnection(object : InstallReferrerStateListener {
    override fun onInstallReferrerSetupFinished(responseCode: Int) {
        if (responseCode == InstallReferrerClient.InstallReferrerResponseLevel.OK) {
            val response: ReferrerDetails = referrerClient.installReferrer
            val referrerUrl = response.installReferrer // e.g., "utm_source=web&utm_campaign=chef_sarah"
            
            // Extract the creator ID from the string
            val creatorId = referrerUrl.split("&")
                .find { it.startsWith("utm_campaign=") }
                ?.substringAfter("utm_campaign=") ?: "organic"
            
            if (creatorId != "organic") {
                saveReferralToFirebase(creatorId)
            }
        }
    }
    override fun onInstallReferrerServiceDisconnected() {
        // Try to restart the connection later if needed
    }
})

3. Firebase Architecture (Firestore + Cloud Functions)
To prevent users from faking downloads and messing up your advocate metrics, use a simple atomic security check in Firebase Cloud Functions.
The Schema: Create a conversions collection where the document ID is the user's unique Android device ID or Firebase Auth UID. This ensures a user can only be counted as a referral once. [1]
Atomic Increments: Use Firestore's FieldValue.increment(1) to update the creator's total download count. This prevents data race conditions when multiple people download the app at the exact same time. [1]
javascript
// Firebase Cloud Function (Node.js) to securely track the download
const { initializeApp } = require('firebase-admin/app');
const { getFirestore, FieldValue } = require('firebase-admin/firestore');

initializeApp();
const db = getFirestore();

exports.trackAppInstall = async (req, res) => {
  const { userId, creatorId } = req.body; // Sent securely from the Android app on first launch

  const conversionRef = db.collection('conversions').doc(userId);
  const conversionDoc = await conversionRef.get();

  // Security Check: Ensure this specific device/user hasn't triggered a count before
  if (!conversionDoc.exists) {
    const batch = db.batch();
    
    // 1. Mark this user as converted
    batch.set(conversionRef, { timestamp: FieldValue.serverTimestamp(), referredBy: creatorId });
    
    // 2. Safely increment the advocate's score by 1
    const creatorRef = db.collection('creators').doc(creatorId);
    batch.update(creatorRef, { total_installs: FieldValue.increment(1) });
    
    await batch.commit();
    return res.status(200).send({ success: true });
  }
  
  return res.status(400).send({ error: "Device already registered" });
};
